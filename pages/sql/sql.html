List Partitioning

RANGE Partitioning

CREATE TABLE emp ( id NUMBER , name VARCHAR2(30) , salary NUMBER ) PARTITION BY RANGE (salary) ( PARTITION low VALUES LESS THAN (10000) , PARTITION medium VALUES LESS THAN (20000) , PARTITION high VALUES LESS THAN (30000) , PARTITION others VALUES LESS THAN (maxvalue) );
