-- If the Data not in the Table is Not Error.

SELECT employee_id, first_name, hire_date, salary, department_id FROM employees
WHERE first_name IN ('Steven', 'Lex', 'Michael', 'NULL');

Note: if we search NULL in IN it doesnot process and fetch that NULL data for example in database there is employee details given as NULL and we search the data using IN. IN doesnot process the data.

SELECT * FROM employees
WHERE department_id IN (10, 60, 90, NULL);
--In IN does not process the NULL.

SORTING

SELECT first_name, last_name FROM employees
ORDER BY first_name ASC;

SELECT first_name, last_name FROM employees
ORDER BY first_name DESC;

SELECT first_name, last_name FROM employees
ORDER BY first_name;

--Postional Based Sorting
SELECT * FROM employees
ORDER BY 4 ASC;

-- Here Salary is multipled by 12 and gave name Alias and order by Alias.
SELECT first_name, salary, 12*salary CTC FROM employees
ORDER BY CTC ASC;

--Sorting 2 Columns 1 ASC and another DeSc bcs without mention sort by ASC.
SELECT employee_id, first_name, hire_date, department_id, salary 
FROM employees
ORDER BY department_id, salary DESC;

SELECT first_name, department_id, salary FROM employees
ORDER BY department_id ASC, salary DESC;

SELECT employee_id, first_name, hire_date, department_id, salary 
FROM employees
ORDER BY department_id, salary;

SELECT employee_id, first_name, hire_date, department_id
FROM employees
ORDER BY department_id DESC;

-- while DESC null will get priority to avoid this NULL will be first.
SELECT employee_id, first_name, hire_date, department_id
FROM employees
ORDER BY department_id DESC NULLS LAST;

SELECT employee_id, first_name, hire_date, department_id
FROM employees
ORDER BY department_id ASC NULLS FIRST;
