DECLARE
   -- Declaration Section
   v_name VARCHAR2(50);
   v_age  NUMBER;

BEGIN
   -- Execution Section
   v_name := 'John';
   v_age  := 30;
   DBMS_OUTPUT.PUT_LINE('Name: ' || v_name);
   DBMS_OUTPUT.PUT_LINE('Age: ' || v_age);

EXCEPTION
   WHEN OTHERS THEN
      DBMS_OUTPUT.PUT_LINE('An error occurred');
END;
